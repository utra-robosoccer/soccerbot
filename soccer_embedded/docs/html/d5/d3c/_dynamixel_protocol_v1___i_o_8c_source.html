142 static inline uint8_t Dynamixel_ComputeChecksum(uint8_t *arr, int length){
143  uint8_t accumulate = 0;
144 
145  /* Loop through the array starting from the 2nd element of the array and
146  * finishing before the last since the last is where the checksum will
147  * be stored */
148  for(uint8_t i = 2; i < length - 1; i++){
149  accumulate += arr[i];
150  }
151 
152  return (~accumulate) & 0xFF; // Lower 8 bits of the logical NOT of the sum
153 }

489 uint16_t Dynamixel_DataReader(
490  Dynamixel_HandleTypeDef* hdynamixel,
491  uint8_t readAddr,
492  uint8_t readLength
493 )
494 {
495  uint16_t retval;
496  uint8_t rxPacketSize;
497  uint8_t recvChecksum;
498  uint8_t computedChecksum;
499  uint8_t ID = hdynamixel -> _ID;
500 
501  if(ID == BROADCAST_ID){
502  ID = 0;
503  }
504 
505  // Do assignments and computations
506  arrTransmit[ID][3] = 4; // Length of message minus the obligatory bytes
507  arrTransmit[ID][4] = INST_READ_DATA; // READ DATA instruction
508  arrTransmit[ID][5] = readAddr; // Write address for register
509  arrTransmit[ID][6] = readLength; // Number of bytes to be read from motor
510  arrTransmit[ID][7] = Dynamixel_ComputeChecksum(arrTransmit[ID], 8);
511 
512  // Are 1 or 2 bytes to be read?
513  rxPacketSize = (readLength == 1) ? 7 : 8;
514 
515  // Set data direction for transmit
516  Dynamixel_BusDirTX(hdynamixel -> _dataDirPort, hdynamixel -> _dataDirPinNum);
517 
518  // Transmit read request
519  if(!Dynamixel_GenericTransmit(hdynamixel, arrTransmit[ID], 8)){
520  hdynamixel -> _lastReadIsValid = false;
521  return -1;
522  }
523 
524  // Receive requested data
525  if(!Dynamixel_GenericReceive(hdynamixel, arrReceive[ID], rxPacketSize)){
526  hdynamixel -> _lastReadIsValid = false;
527  return -1;
528  }
529 
530  // Check data integrity and place this flag in a field the application can read
531  recvChecksum = arrReceive[ID][rxPacketSize - 1];
532  computedChecksum = Dynamixel_ComputeChecksum(arrReceive[ID], rxPacketSize);
533  hdynamixel -> _lastReadIsValid = (computedChecksum == recvChecksum);
534 
535  retval = (uint16_t)arrReceive[ID][5];
536  if(readLength == 2){
537  retval |= arrReceive[ID][6] << 8;
538  }
539 
540  return retval;
541 }

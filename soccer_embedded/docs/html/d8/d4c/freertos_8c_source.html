680  // designer we are using does not allow us to generate such filters in
681  // the free version, so this is the best we can do unless we use other
682  // software.
683  if (i % 16 == 0) {
684  MPU6050_Read_Gyroscope_Withoffset_IT(&IMUdata);
685  MPUFilter_FilterAngularVelocity(&IMUdata);
686  }
687  i++;
688 
689  dataToSend.pData = &IMUdata;
690  xQueueSend(TXQueueHandle, &dataToSend, 0);
691  }

704 void StartRxTask(void const * argument)
705 {
706  uint8_t robotGoalData[sizeof(RobotGoal)];
707  uint8_t *robotGoalDataPtr;
708  uint8_t buffRx[92];
709  uint8_t startSeqCount;
710  uint8_t totalBytesRead;
711 
712  // Receiving
713  robotGoal.id = 0;
714  robotGoalDataPtr = robotGoalData;
715  startSeqCount = 0;
716  totalBytesRead = 0;
717 
718  // Sending
719  robotState.id = 0;
720  robotState.start_seq = UINT32_MAX;
721  robotState.end_seq = 0;
722 
723  xTaskNotifyWait(UINT32_MAX, UINT32_MAX, NULL, portMAX_DELAY);
724 
725  HAL_StatusTypeDef status;
726 
727  uint32_t notification;
728 
729  HAL_UART_Receive_DMA(&huart5, (uint8_t*)buffRx, sizeof(buffRx));
730 
731  for (;;)
732  {
733  // Wait until notified from ISR. Clear no bits on entry in case the notification
734  // comes before this statement is executed (which is rather unlikely as long as
735  // this task has the highest priority, but overall this is a better decision in
736  // case priorities are changed in the future and someone forgets about this.
737  do{
738  xTaskNotifyWait(0, NOTIFIED_FROM_RX_ISR, &notification, portMAX_DELAY);
739  }while((notification & NOTIFIED_FROM_RX_ISR) != NOTIFIED_FROM_RX_ISR);
740 
741  do{
742  // This do-while loop with the mutex inside of it makes calls to the UART module
743  // responsible for PC communication atomic. This attempts to solve the following
744  // scenario: the TX thread is in the middle of executing the call to HAL_UART_Transmit
745  // when suddenly the RX thread is unblocked. The RX thread calls HAL_UART_Receive, and
746  // returns immediately when it detects that the uart module is already locked. Then
747  // the RX thread blocks itself and never wakes up since a RX transfer was never
748  // initialized.
749  xSemaphoreTake(PCUARTHandle, 1);
750  status = HAL_UART_Receive_DMA(&huart5, (uint8_t*)buffRx, sizeof(buffRx));
751  xSemaphoreGive(PCUARTHandle);
752  }while(status != HAL_OK);
753 
754  for (uint8_t i = 0; i < sizeof(buffRx); i++) {
755  if (startSeqCount == 4) {
756  // This control block is entered when the header sequence of
757  // 0xFFFFFFFF has been received; thus we know the data we
758  // receive will be in tact
759 
760  *robotGoalDataPtr = buffRx[i];
761  robotGoalDataPtr++;
762  totalBytesRead++;
763 
764  if (totalBytesRead == sizeof(RobotGoal)) {
765  // If, after the last couple of receive interrupts, we have
766  // received sizeof(RobotGoal) bytes, then we copy the data
767  // buffer into the robotGoal structure and wake the control
768  // thread to distribute states to each actuator
769  memcpy(&robotGoal, robotGoalData, sizeof(RobotGoal));
770  robotState.id = robotGoal.id;
771 
772  // Reset the variables to help with reception of a RobotGoal
773  robotGoalDataPtr = robotGoalData;
774  startSeqCount = 0;
775  totalBytesRead = 0;
776 
777  xTaskNotify(CommandTaskHandle, NOTIFIED_FROM_TASK, eSetBits); // Wake control task
778  xTaskNotify(IMUTaskHandle, NOTIFIED_FROM_TASK, eSetBits); // Wake MPU task
779  continue;
780  }
781  }else{
782  // This control block is used to verify that the data header is in tact
783  if (buffRx[i] == 0xFF) {
784  startSeqCount++;
785  } else {
786  startSeqCount = 0;
787  }
788  }
789  }
790  }
791 }

805 void StartTxTask(void const * argument)
806 {
807  xTaskNotifyWait(UINT32_MAX, UINT32_MAX, NULL, portMAX_DELAY);
808 
809  TXData_t receivedData;
810  Dynamixel_HandleTypeDef* motorPtr = NULL;
811  MPU6050_HandleTypeDef* imuPtr = NULL;
812  char* const pIMUXGyroData = &robotState.msg[ROBOT_STATE_MPU_DATA_OFFSET];
813 
814  HAL_StatusTypeDef status;
815  uint32_t notification;
816  uint32_t dataReadyFlags = 0; // Bits in this are set based on which sensor data is ready
817 
818  uint32_t NOTIFICATION_MASK = 0x80000000;
819  for(uint8_t i = 1; i <= 12; i++){
820  NOTIFICATION_MASK |= (1 << i);
821  }
822 
823  for(;;)
824  {
825  while((dataReadyFlags & NOTIFICATION_MASK) != NOTIFICATION_MASK){
826  while(xQueueReceive(TXQueueHandle, &receivedData, portMAX_DELAY) != pdTRUE);
827 
828  switch(receivedData.eDataType){
829  case eMotorData:
830  motorPtr = (Dynamixel_HandleTypeDef*)receivedData.pData;
831 
832  if(motorPtr == NULL){ break; }
833 
834  // Validate data and store it in robotState
835  if(motorPtr->_ID <= NUM_MOTORS){
836  // Copy sensor data for this motor into its section of robotState.msg
837  memcpy(&robotState.msg[4 * (motorPtr->_ID - 1)], &(motorPtr->_lastPosition), sizeof(float));
838 
839  // Set flag indicating the motor with this id has reported in with position data
840  dataReadyFlags |= (1 << motorPtr->_ID);
841  }
842  break;
843  case eIMUData:
844  imuPtr = (MPU6050_HandleTypeDef*)receivedData.pData;
845 
846  if(imuPtr == NULL){ break; }
847 
848  // Copy sensor data into the IMU data section of robotState.msg
849  memcpy(pIMUXGyroData, (&imuPtr->_X_GYRO), 6 * sizeof(float));
850 
851  // Set flag indicating IMU data has reported in
852  dataReadyFlags |= 0x80000000;
853  break;
854  default:
855  break;
856  }
857  }
858  dataReadyFlags = 0; // Clear all flags
859 
860  do{
861  // This do-while loop with the mutex inside of it makes calls to the UART module
862  // responsible for PC communication atomic. This attempts to solve the following
863  // scenario: the TX thread is in the middle of executing the call to HAL_UART_Transmit
864  // when suddenly the RX thread is unblocked. The RX thread calls HAL_UART_Receive, and
865  // returns immediately when it detects that the uart module is already locked. Then
866  // the RX thread blocks itself and never wakes up since a RX transfer was never
867  // initialized.
868  xSemaphoreTake(PCUARTHandle, 1);
869  status = HAL_UART_Transmit_DMA(&huart5, (uint8_t*)&robotState, sizeof(RobotState));
870  xSemaphoreGive(PCUARTHandle);
871  }while(status != HAL_OK);
872 
873  // Wait until notified from ISR. Clear no bits on entry in case the notification
874  // came while a higher priority task was executing.
875  do{
876  xTaskNotifyWait(0, NOTIFIED_FROM_TX_ISR, &notification, portMAX_DELAY);
877  }while((notification & NOTIFIED_FROM_TX_ISR) != NOTIFIED_FROM_TX_ISR);
878  }
879 }

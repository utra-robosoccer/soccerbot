95  static struct I2C_Module{
96  I2C_HandleTypeDef* instance;
97  uint16_t sdaPin;
98  GPIO_TypeDef* sdaPort;
99  uint16_t sclPin;
100  GPIO_TypeDef* sclPort;
101  }i2cmodule = {&hi2c1, GPIO_PIN_7, GPIOB, GPIO_PIN_6, GPIOB};
102  static struct I2C_Module* i2c = &i2cmodule;
103  static uint8_t timeout = 1;
104 
105  GPIO_InitTypeDef GPIO_InitStructure;
106 
107  I2C_HandleTypeDef* handler = NULL;
108 
109  handler = i2c->instance;
110 
111  // 1. Clear PE bit.
112  CLEAR_BIT(handler->Instance->CR1, I2C_CR1_PE);
113 
114  GPIO_InitStructure.Mode = GPIO_MODE_OUTPUT_OD;
115  GPIO_InitStructure.Pull = GPIO_NOPULL;
116  GPIO_InitStructure.Speed = GPIO_SPEED_FREQ_VERY_HIGH;
117 
118  GPIO_InitStructure.Pin = i2c->sclPin;
119  HAL_GPIO_Init(i2c->sclPort, &GPIO_InitStructure);
120 
121  GPIO_InitStructure.Pin = i2c->sdaPin;
122  HAL_GPIO_Init(i2c->sdaPort, &GPIO_InitStructure);
123 
124  for(uint8_t i = 0; i < numClocks; i++){
125  // 3. Check SCL and SDA High level in GPIOx_IDR.
126  if(sendStopBits){HAL_GPIO_WritePin(i2c->sdaPort, i2c->sdaPin, GPIO_PIN_SET);}
127  HAL_GPIO_WritePin(i2c->sclPort, i2c->sclPin, GPIO_PIN_SET);
128 
129  wait_for_gpio_state_timeout(i2c->sclPort, i2c->sclPin, GPIO_PIN_SET, timeout);
130  if(sendStopBits){wait_for_gpio_state_timeout(i2c->sdaPort, i2c->sdaPin, GPIO_PIN_SET, timeout);}
131 
132  // 4. Configure the SDA I/O as General Purpose Output Open-Drain, Low level (Write 0 to GPIOx_ODR).
133  if(sendStopBits){
134  HAL_GPIO_WritePin(i2c->sdaPort, i2c->sdaPin, GPIO_PIN_RESET);
135  wait_for_gpio_state_timeout(i2c->sdaPort, i2c->sdaPin, GPIO_PIN_RESET, timeout); // 5. Check SDA Low level in GPIOx_IDR
136  }
137 
138  // 6. Configure the SCL I/O as General Purpose Output Open-Drain, Low level (Write 0 to GPIOx_ODR).
139  HAL_GPIO_WritePin(i2c->sclPort, i2c->sclPin, GPIO_PIN_RESET);
140  wait_for_gpio_state_timeout(i2c->sclPort, i2c->sclPin, GPIO_PIN_RESET, timeout); // 7. Check SCL Low level in GPIOx_IDR.
141 
142  // 8. Configure the SCL I/O as General Purpose Output Open-Drain, High level (Write 1 to GPIOx_ODR).
143  HAL_GPIO_WritePin(i2c->sclPort, i2c->sclPin, GPIO_PIN_SET);
144  wait_for_gpio_state_timeout(i2c->sclPort, i2c->sclPin, GPIO_PIN_SET, timeout); // 9. Check SCL High level in GPIOx_IDR.
145 
146  // 10. Configure the SDA I/O as General Purpose Output Open-Drain , High level (Write 1 to GPIOx_ODR).
147  if(sendStopBits){
148  HAL_GPIO_WritePin(i2c->sdaPort, i2c->sdaPin, GPIO_PIN_SET);
149  wait_for_gpio_state_timeout(i2c->sdaPort, i2c->sdaPin, GPIO_PIN_SET, timeout); // 11. Check SDA High level in GPIOx_IDR.
150  }
151  }
152 
153  // 12. Configure the SCL and SDA I/Os as Alternate function Open-Drain.
154  GPIO_InitStructure.Mode = GPIO_MODE_AF_OD;
155  GPIO_InitStructure.Alternate = GPIO_AF4_I2C1;
156 
157  GPIO_InitStructure.Pin = i2c->sclPin;
158  HAL_GPIO_Init(i2c->sclPort, &GPIO_InitStructure);
159 
160  GPIO_InitStructure.Pin = i2c->sdaPin;
161  HAL_GPIO_Init(i2c->sdaPort, &GPIO_InitStructure);
162 
163  // 13. Set SWRST bit in I2Cx_CR1 register.
164  SET_BIT(handler->Instance->CR1, I2C_CR1_SWRST);
165  asm("nop");
166 
167  /* 14. Clear SWRST bit in I2Cx_CR1 register. */
168  CLEAR_BIT(handler->Instance->CR1, I2C_CR1_SWRST);
169  asm("nop");
170 
171  /* 15. Enable the I2C peripheral by setting the PE bit in I2Cx_CR1 register */
172  SET_BIT(handler->Instance->CR1, I2C_CR1_PE);
173  asm("nop");
174

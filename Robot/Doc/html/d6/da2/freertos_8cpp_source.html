685  // delays, we also use a filter with fewer taps than the acceleration
686  // filters. Ideally: we would sample faster to reduce aliasing, then
687  // use a filter with a smaller cutoff frequency. However, the filter
688  // designer we are using does not allow us to generate such filters in
689  // the free version, so this is the best we can do unless we use other
690  // software.
691  if (i % 16 == 0) {
692  MPU6050_Read_Gyroscope_Withoffset_IT(&IMUdata);
693  MPUFilter_FilterAngularVelocity(&IMUdata);
694  }
695  i++;
696 
697  dataToSend.pData = &IMUdata;
698  xQueueSend(TXQueueHandle, &dataToSend, 0);
699  }

712 void StartRxTask(void const * argument)
713 {
714  uint8_t robotGoalData[sizeof(RobotGoal)];
715  uint8_t *robotGoalDataPtr;
716  uint8_t buffRx[92];
717  uint8_t startSeqCount;
718  uint8_t totalBytesRead;
719 
720  // Receiving
721  robotGoal.id = 0;
722  robotGoalDataPtr = robotGoalData;
723  startSeqCount = 0;
724  totalBytesRead = 0;
725 
726  // Sending
727  robotState.id = 0;
728  robotState.start_seq = UINT32_MAX;
729  robotState.end_seq = 0;
730 
731  xTaskNotifyWait(UINT32_MAX, UINT32_MAX, NULL, portMAX_DELAY);
732 
733  HAL_StatusTypeDef status;
734 
735  uint32_t notification;
736 
737  HAL_UART_Receive_DMA(&huart5, (uint8_t*)buffRx, sizeof(buffRx));
738 
739  for (;;)
740  {
741  // Wait until notified from ISR. Clear no bits on entry in case the notification
742  // comes before this statement is executed (which is rather unlikely as long as
743  // this task has the highest priority, but overall this is a better decision in
744  // case priorities are changed in the future and someone forgets about this.
745  do{
746  xTaskNotifyWait(0, NOTIFIED_FROM_RX_ISR, &notification, portMAX_DELAY);
747  }while((notification & NOTIFIED_FROM_RX_ISR) != NOTIFIED_FROM_RX_ISR);
748 
749  do{
750  // This do-while loop with the mutex inside of it makes calls to the UART module
751  // responsible for PC communication atomic. This attempts to solve the following
752  // scenario: the TX thread is in the middle of executing the call to HAL_UART_Transmit
753  // when suddenly the RX thread is unblocked. The RX thread calls HAL_UART_Receive, and
754  // returns immediately when it detects that the uart module is already locked. Then
755  // the RX thread blocks itself and never wakes up since a RX transfer was never
756  // initialized.
757  xSemaphoreTake(PCUARTHandle, 1);
758  status = HAL_UART_Receive_DMA(&huart5, (uint8_t*)buffRx, sizeof(buffRx));
759  xSemaphoreGive(PCUARTHandle);
760  }while(status != HAL_OK);
761 
762  for (uint8_t i = 0; i < sizeof(buffRx); i++) {
763  if (startSeqCount == 4) {
764  // This control block is entered when the header sequence of
765  // 0xFFFFFFFF has been received; thus we know the data we
766  // receive will be in tact
767 
768  *robotGoalDataPtr = buffRx[i];
769  robotGoalDataPtr++;
770  totalBytesRead++;
771 
772  if (totalBytesRead == sizeof(RobotGoal)) {
773  // If, after the last couple of receive interrupts, we have
774  // received sizeof(RobotGoal) bytes, then we copy the data
775  // buffer into the robotGoal structure and wake the control
776  // thread to distribute states to each actuator
777  memcpy(&robotGoal, robotGoalData, sizeof(RobotGoal));
778  robotState.id = robotGoal.id;
779 
780  // Reset the variables to help with reception of a RobotGoal
781  robotGoalDataPtr = robotGoalData;
782  startSeqCount = 0;
783  totalBytesRead = 0;
784 
785  xTaskNotify(CommandTaskHandle, NOTIFIED_FROM_TASK, eSetBits); // Wake control task
786  xTaskNotify(IMUTaskHandle, NOTIFIED_FROM_TASK, eSetBits); // Wake MPU task
787  continue;
788  }
789  }else{
790  // This control block is used to verify that the data header is in tact
791  if (buffRx[i] == 0xFF) {
792  startSeqCount++;
793  } else {
794  startSeqCount = 0;
795  }
796  }
797  }
798  }
799 }

813 void StartTxTask(void const * argument)
814 {
815  xTaskNotifyWait(UINT32_MAX, UINT32_MAX, NULL, portMAX_DELAY);
816 
817  TXData_t receivedData;
818  Dynamixel_HandleTypeDef* motorPtr = NULL;
819  MPU6050_HandleTypeDef* imuPtr = NULL;
820  char* const pIMUXGyroData = &robotState.msg[ROBOT_STATE_MPU_DATA_OFFSET];
821 
822  HAL_StatusTypeDef status;
823  uint32_t notification;
824  uint32_t dataReadyFlags = 0; // Bits in this are set based on which sensor data is ready
825 
826  uint32_t NOTIFICATION_MASK = 0x80000000;
827  for(uint8_t i = 1; i <= 12; i++){
828  NOTIFICATION_MASK |= (1 << i);
829  }
830 
831  for(;;)
832  {
833  while((dataReadyFlags & NOTIFICATION_MASK) != NOTIFICATION_MASK){
834  while(xQueueReceive(TXQueueHandle, &receivedData, portMAX_DELAY) != pdTRUE);
835 
836  switch(receivedData.eDataType){
837  case eMotorData:
838  motorPtr = (Dynamixel_HandleTypeDef*)receivedData.pData;
839 
840  if(motorPtr == NULL){ break; }
841 
842  // Validate data and store it in robotState
843  if(motorPtr->_ID <= NUM_MOTORS){
844  // Copy sensor data for this motor into its section of robotState.msg
845  memcpy(&robotState.msg[4 * (motorPtr->_ID - 1)], &(motorPtr->_lastPosition), sizeof(float));
846 
847  // Set flag indicating the motor with this id has reported in with position data
848  dataReadyFlags |= (1 << motorPtr->_ID);
849  }
850  break;
851  case eIMUData:
852  imuPtr = (MPU6050_HandleTypeDef*)receivedData.pData;
853 
854  if(imuPtr == NULL){ break; }
855 
856  // Copy sensor data into the IMU data section of robotState.msg
857  memcpy(pIMUXGyroData, (&imuPtr->_X_GYRO), 6 * sizeof(float));
858 
859  // Set flag indicating IMU data has reported in
860  dataReadyFlags |= 0x80000000;
861  break;
862  default:
863  break;
864  }
865  }
866  dataReadyFlags = 0; // Clear all flags
867 
868  do{
869  // This do-while loop with the mutex inside of it makes calls to the UART module
870  // responsible for PC communication atomic. This attempts to solve the following
871  // scenario: the TX thread is in the middle of executing the call to HAL_UART_Transmit
872  // when suddenly the RX thread is unblocked. The RX thread calls HAL_UART_Receive, and
873  // returns immediately when it detects that the uart module is already locked. Then
874  // the RX thread blocks itself and never wakes up since a RX transfer was never
875  // initialized.
876  xSemaphoreTake(PCUARTHandle, 1);
877  status = HAL_UART_Transmit_DMA(&huart5, (uint8_t*)&robotState, sizeof(RobotState));
878  xSemaphoreGive(PCUARTHandle);
879  }while(status != HAL_OK);
880 
881  // Wait until notified from ISR. Clear no bits on entry in case the notification
882  // came while a higher priority task was executing.
883  do{
884  xTaskNotifyWait(0, NOTIFIED_FROM_TX_ISR, &notification, portMAX_DELAY);
885  }while((notification & NOTIFIED_FROM_TX_ISR) != NOTIFIED_FROM_TX_ISR);
886  }
887 }

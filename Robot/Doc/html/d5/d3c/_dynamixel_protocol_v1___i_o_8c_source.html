146 static inline uint8_t Dynamixel_ComputeChecksum(uint8_t *arr, int length){
147  uint8_t accumulate = 0;
148 
149  /* Loop through the array starting from the 2nd element of the array and
150  * finishing before the last since the last is where the checksum will
151  * be stored */
152  for(uint8_t i = 2; i < length - 1; i++){
153  accumulate += arr[i];
154  }
155 
156  return (~accumulate) & 0xFF; // Lower 8 bits of the logical NOT of the sum
157 }

407  uint8_t readLength

415

418  }
419 
420  // Do assignments and computations

422  arrTransmit[ID][4] = INST_READ_DATA; // READ DATA instruction
423  arrTransmit[ID][5] = readAddr; // Write address for register
424  arrTransmit[ID][6] = readLength; // Number of bytes to be read from motor
425  arrTransmit[ID][7] = Dynamixel_ComputeChecksum(arrTransmit[ID], 8);
426 
427  // Are 1 or 2 bytes to be read?
428  rxPacketSize = (readLength == 1) ? 7 : 8;
429 
430  // Set data direction for transmit
431  Dynamixel_BusDirTX(hdynamixel -> _dataDirPort, hdynamixel -> _dataDirPinNum);
432 
433  // Transmit read request
434  if(!Dynamixel_GenericTransmit(hdynamixel, arrTransmit[ID], 8)){
435  hdynamixel -> _lastReadIsValid = false;
436  return -1;
437  }
438 
439  // Receive requested data
440  if(!Dynamixel_GenericReceive(hdynamixel, arrReceive[ID], rxPacketSize)){
441  hdynamixel -> _lastReadIsValid = false;
442  return -1;
443  }
444 
445  // Check data integrity and place this flag in a field the application can read
446  recvChecksum = arrReceive[ID][rxPacketSize - 1];
447  computedChecksum = Dynamixel_ComputeChecksum(arrReceive[ID], rxPacketSize);
448  hdynamixel -> _lastReadIsValid = (computedChecksum == recvChecksum);
449 
450  retval = (uint16_t)arrReceive[ID][5];
451  if(readLength == 2){
452  retval |= arrReceive[ID][6] << 8;
453  }
